const COMM_SHEET = 'Commercial'; const EXPR_SHEET = 'Express'; function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var name = (e.parameter.sheet === 'express') ? EXPR_SHEET : COMM_SHEET; var ws = ss.getSheetByName(name); var data = ws ? ws.getDataRange().getValues() : []; return ContentService .createTextOutput(JSON.stringify({ok:true, data:data})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var body = JSON.parse(e.postData.contents); var name = (body.sheet === 'express') ? EXPR_SHEET : COMM_SHEET; var ws = ss.getSheetByName(name); if (!ws) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:'Sheet not found'})) .setMimeType(ContentService.MimeType.JSON); } if (body.action === 'append') { ws.appendRow(body.row); } else if (body.action === 'update') { ws.getRange(body.rowIndex, 1, 1, body.row.length).setValues([body.row]); } else if (body.action === 'delete') { ws.deleteRow(body.rowIndex); } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
