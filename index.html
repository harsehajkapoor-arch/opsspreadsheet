const COMM_SHEET = 'Commercial'; const EXPR_SHEET = 'Express'; function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const name = (e.parameter.sheet === 'express') ? EXPR_SHEET : COMM_SHEET; const ws = ss.getSheetByName(name); const data = ws ? ws.getDataRange().getValues() : []; return ContentService .createTextOutput(JSON.stringify({ok:true, data:data})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const body = JSON.parse(e.postData.contents); const name = (body.sheet === 'express') ? EXPR_SHEET : COMM_SHEET; const ws = ss.getSheetByName(name); if (!ws) return resp({ok:false,error:'Sheet not found'}); if (body.action === 'append') ws.appendRow(body.row); else if (body.action === 'update') ws.getRange(body.rowIndex,1,1,body.row.length).setValues([body.row]); else if (body.action === 'delete') ws.deleteRow(body.rowIndex); return resp({ok:true}); } function resp(o){return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON);}
