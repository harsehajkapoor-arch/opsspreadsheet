Step 1 — Deploy Your Free Backend (2 minutes)

1
Open your Google Sheet → Extensions → Apps Script

A code editor will open. Delete any existing code.

2
Paste this entire script and click Save (Ctrl+S)
const SHEET_ID = 'YOUR_SHEET_ID_HERE'; // ← paste your sheet ID here const COMM_SHEET = 'Commercial'; const EXPR_SHEET = 'Express'; function doGet(e) { const sheet = e.parameter.sheet === 'express' ? EXPR_SHEET : COMM_SHEET; const ss = SpreadsheetApp.openById(SHEET_ID); const ws = ss.getSheetByName(sheet); const data = ws.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify({ok:true,data})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const body = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); const sheetName = body.sheet === 'express' ? EXPR_SHEET : COMM_SHEET; const ws = ss.getSheetByName(sheetName); if(body.action === 'append') { ws.appendRow(body.row); } else if(body.action === 'update') { const range = ws.getRange(body.rowIndex, 1, 1, body.row.length); range.setValues([body.row]); } else if(body.action === 'delete') { ws.deleteRow(body.rowIndex); } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
